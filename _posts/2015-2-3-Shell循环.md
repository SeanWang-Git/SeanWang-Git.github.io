#### for-in-do-done

	eg:
		for tmp in tmp_array
		do
			command
		done

	tmp 可以为数组，数字，字符串
	
	#!/bin/bash

	tmp_array=(1 2 3 4 5 6)

	tmp_array_str=(
	'A'
	'B'
	'C')
	tmp_str="This is my style"
	
	for tmp in ${tmp_array[*]}
	do
	        echo ${tmp}
	done
	
	for tmp in ${tmp_array_str[@]}
	do
	        echo ${tmp}
	done
	
	for tmp in ${tmp_str}
	do
	        echo ${tmp}
	done
	
	for tmp in /home/*
	do
	        echo ${tmp}
	done
	
#### while-do-done

	eg:
		while expression
		do 
			command
		done

	#!/bin/bash

	num=1

	while [ ${num} -le 10 ]
	do
	        echo ${num}
	        num=`expr ${num} + 1`
	done

#### until-do-done

	eg:
		until expression
		do
			command
		done

	#!/bin/bash
	num=1
	until [ ${num} -ge 10 ]
	do
		echo ${num}
		num=`expr ${num} + 1`
	done
